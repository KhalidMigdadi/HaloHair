CREATE DATABASE BarberBookingDB;
USE BarberBookingDB;

-- Users (Customers)
CREATE TABLE Users (
    UserID INT IDENTITY(1,1) PRIMARY KEY,
    FullName NVARCHAR(100) NOT NULL,
    Email NVARCHAR(100) UNIQUE NOT NULL,
    PhoneNumber NVARCHAR(20) NOT NULL,
    PasswordHash NVARCHAR(255) NOT NULL,
    JoinDate DATETIME DEFAULT GETDATE()
);

-- Barbers
CREATE TABLE Barbers (
    BarberID INT IDENTITY(1,1) PRIMARY KEY,
    FullName NVARCHAR(100) NOT NULL,
    Specialization NVARCHAR(100),
    ExperienceYears INT,
    PhoneNumber NVARCHAR(20),
    AvailableStatus BIT DEFAULT 1
);

-- Services (Haircuts, beard trims, etc.)
CREATE TABLE Services (
    ServiceID INT IDENTITY(1,1) PRIMARY KEY,
    ServiceName NVARCHAR(100) NOT NULL,
    Price DECIMAL(10,2) NOT NULL,
    Duration INT NOT NULL -- Duration in minutes
);

-- Barber Services (Which barber provides which service)
CREATE TABLE BarberServices (
    BarberServiceID INT IDENTITY(1,1) PRIMARY KEY,
    BarberID INT FOREIGN KEY REFERENCES Barbers(BarberID) ON DELETE CASCADE,
    ServiceID INT FOREIGN KEY REFERENCES Services(ServiceID) ON DELETE CASCADE,
    Price DECIMAL(10,2) NOT NULL
);

-- Appointments (Booking records)
CREATE TABLE Appointments (
    AppointmentID INT IDENTITY(1,1) PRIMARY KEY,
    UserID INT FOREIGN KEY REFERENCES Users(UserID) ON DELETE CASCADE,
    BarberID INT FOREIGN KEY REFERENCES Barbers(BarberID) ON DELETE CASCADE,
    ServiceID INT FOREIGN KEY REFERENCES Services(ServiceID) ON DELETE CASCADE,
    AppointmentDate DATETIME NOT NULL,
    Status NVARCHAR(50) CHECK (Status IN ('Pending', 'Confirmed', 'Canceled')) DEFAULT 'Pending'
);

-- Payments
CREATE TABLE Payments (
    PaymentID INT IDENTITY(1,1) PRIMARY KEY,
    UserID INT FOREIGN KEY REFERENCES Users(UserID) ON DELETE CASCADE,
    AppointmentID INT FOREIGN KEY REFERENCES Appointments(AppointmentID) ON DELETE CASCADE,
    Amount DECIMAL(10,2) NOT NULL,
    PaymentDate DATETIME DEFAULT GETDATE(),
    Status NVARCHAR(50) CHECK (Status IN ('Paid', 'Pending', 'Failed')) DEFAULT 'Pending'
);

-- Working Hours (Barber Availability)
CREATE TABLE WorkingHours (
    WorkingHourID INT IDENTITY(1,1) PRIMARY KEY,
    BarberID INT FOREIGN KEY REFERENCES Barbers(BarberID) ON DELETE CASCADE,
    DayOfWeek NVARCHAR(20) NOT NULL, -- Monday, Tuesday, etc.
    StartTime TIME NOT NULL,
    EndTime TIME NOT NULL
);

-- Reviews (Customer feedback)
CREATE TABLE Reviews (
    ReviewID INT IDENTITY(1,1) PRIMARY KEY,
    UserID INT FOREIGN KEY REFERENCES Users(UserID) ON DELETE CASCADE,
    BarberID INT FOREIGN KEY REFERENCES Barbers(BarberID) ON DELETE CASCADE,
    Rating INT CHECK (Rating BETWEEN 1 AND 5),
    Comment NVARCHAR(500),
    ReviewDate DATETIME DEFAULT GETDATE()
);

-- Notifications (Booking reminders)
CREATE TABLE Notifications (
    NotificationID INT IDENTITY(1,1) PRIMARY KEY,
    UserID INT FOREIGN KEY REFERENCES Users(UserID) ON DELETE CASCADE,
    Message NVARCHAR(255) NOT NULL,
    IsRead BIT DEFAULT 0,
    CreatedAt DATETIME DEFAULT GETDATE()
);

-- Loyalty Points (Rewards for customers)
CREATE TABLE LoyaltyPoints (
    LoyaltyPointID INT IDENTITY(1,1) PRIMARY KEY,
    UserID INT FOREIGN KEY REFERENCES Users(UserID) ON DELETE CASCADE,
    Points INT DEFAULT 0,
    LastUpdated DATETIME DEFAULT GETDATE()
);
